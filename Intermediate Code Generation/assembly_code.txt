	PUSH AX
	PUSH BX

	PUSH DX

	LEA SI,NUMBER
	MOV BX,10
	ADD SI,4
	CMP AX,0
	JNGE NEGATE
	PRINT:
	XOR DX,DX
	DIV BX
	MOV [SI],DL
	ADD [SI],'0'
	DEC SI
	CMP AX,0
	JNE PRINT
	INC SI
	LEA DX,SI
	MOV AH,9
	INT 21H
	POP SI
	POP DX
	POP CX
	POP BX
	POP AX
	RET
	NEGATE:

	MOV AH,2
	MOV DL,'-'
	INT 21H
	POP AX
	NEG AX
	JMP PRINT
